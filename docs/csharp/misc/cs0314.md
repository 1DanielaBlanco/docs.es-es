## <a name="example"></a><span data-ttu-id="bb9fc-108">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="bb9fc-108">Example</span></span>

// cs0314.cs

public class ClassConstraint { }  
  
public class A<T> where T : ClassConstraint  
{ }

{ }

{ }
