 <span data-ttu-id="d12a6-104">No se permiten rutinas de conversión duplicadas en una clase.</span><span class="sxs-lookup"><span data-stu-id="d12a6-104">Duplicate conversion routines are not allowed in a class.</span></span>

// CS0557.cs

    public class ii  
    {  
        public class iii  
        {  
        public static implicit operator int(iii aa)  
        {  
            return 0;  
        }

        {  
            return 0;  
        }  
        }

        {  
        }  
    }  
}
