```  
// CS0559.cs

{  
   public static implicit operator int(iii x)  
   {  
      return 0;  
   }  
  
   public static implicit operator iii(int x)  
   {  
      return null;  
   }  
  
   public static int operator ++(int aa)   // CS0559  
   // try the following line instead  
   // public static iii operator ++(iii aa)  
   {  
      return (iii)0;  
   }  
}  
```  
  
## <a name="example"></a><span data-ttu-id="a3f28-108">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="a3f28-108">Example</span></span>  
 <span data-ttu-id="a3f28-109">El ejemplo siguiente genera la advertencia CS0559.</span><span class="sxs-lookup"><span data-stu-id="a3f28-109">The following sample generates CS0559.</span></span>  
  
```  
// CS0559_b.cs  
// compile with: /target:library  
public struct S  
{  
   public static S operator ++(S? s) { return new S(); }   // CS0559  
   public static S operator --(S? s) { return new S(); }   // CS0559  
}  
  
public struct T  
{  
// OK  
   public static T operator --(T t) { return new T(); }  
   public static T operator ++(T t) { return new T(); }  
  
   public static T? operator --(T? t) { return new T(); }  
   public static T? operator ++(T? t) { return new T(); }  
}  
```
