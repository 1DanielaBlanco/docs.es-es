## <a name="example"></a><span data-ttu-id="5c24c-109">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="5c24c-109">Example</span></span>

{  
    static void Main()  
    {  
        int[] array = new int[] { 1, 2, 3, 4, 5 };
