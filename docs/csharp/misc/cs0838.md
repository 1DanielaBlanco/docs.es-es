## <a name="example"></a><span data-ttu-id="89177-107">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="89177-107">Example</span></span>

```csharp  
using System.Linq;  
using System.Linq.Expressions;  
  
namespace TestNamespace  
{  
    class Test  
    {  
        static int Main()  
        {

                () => new int[2, 2] { { 1, 2 }, { 3, 4 } }; // CS0838

            int[,] nums = new int[2, 2] { { 1, 2 }, { 3, 4 } };  
            Expression<Func<int[,]>> expr2 = () => nums;   
  
            return 1;  
        }  
    }  
}  
```
